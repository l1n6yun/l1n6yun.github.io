js代码

```javascript
var countdown = 60;

function setTime(val) {
    if (countdown === 0) {
        val.removeAttribute("disabled");
        val.value = "免费获取验证码";
        countdown = 60;
    } else {
        val.setAttribute("disabled", true);
        val.value = "重新发送(" + countdown + ")";
        countdown--;
        setTimeout(function () {
            setTime(val)
        }, 1000)
    }
}
```
